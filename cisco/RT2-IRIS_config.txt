! ============================================================
! RT2-IRIS_config.txt — Cisco ISR 1941W (Routeur + WiFi)
! Projet : IRIS-NICE-2024-RP01
! Auteur  : Nedjmeddine Belloum
!
! Rôle    : Routeur inter-VLAN + NAT + RADIUS (802.1X WiFi)
! Interfaces :
!   Gi0/0   — WAN (DHCP FAI)
!   Gi0/1   — LAN trunk vers SW2-IRIS
!   Wlan0   — AP WiFi intégré
!
! CORRECTION CRITIQUE appliquée :
!   NAT_LIST — ajout de permit 192.168.50.0/24 (VLAN Management)
!   → DC-IRIS-01 peut joindre Internet (Windows Update, activation)
! ============================================================

version 15.2
service timestamps debug datetime msec
service timestamps log datetime msec
service password-encryption
!
hostname RT2-IRIS
!
enable secret NVTech_Admin2026!
!
username admin privilege 15 secret NVTech_Admin2026!
!
! ── AAA (802.1X WiFi via NPS) ─────────────────────────────
aaa new-model
aaa authentication dot1x default group radius
aaa authorization network default group radius
aaa accounting dot1x default start-stop group radius
!
! ── RADIUS Clients ────────────────────────────────────────
radius server NPS-IRIS
 address ipv4 192.168.50.10 auth-port 1812 acct-port 1813
 key RadiusRTR_IRIS_2026!
!
dot1x system-auth-control
!
! ── Interface WAN ─────────────────────────────────────────
interface GigabitEthernet0/0
 description WAN-Internet
 ip address dhcp
 ip nat outside
 ip access-group BLOCK-WAN-IN in
 no shutdown
!
! ── Sous-interfaces LAN (802.1Q inter-VLAN routing) ───────
interface GigabitEthernet0/1
 description TRUNK-vers-SW2-IRIS
 no ip address
 no shutdown
!
interface GigabitEthernet0/1.10
 description VLAN10-ETUDIANTS
 encapsulation dot1Q 10
 ip address 192.168.10.1 255.255.255.0
 ip nat inside
 ip helper-address 192.168.50.10
 no shutdown
!
interface GigabitEthernet0/1.20
 description VLAN20-PROFS
 encapsulation dot1Q 20
 ip address 192.168.20.1 255.255.255.0
 ip nat inside
 ip helper-address 192.168.50.10
 no shutdown
!
interface GigabitEthernet0/1.30
 description VLAN30-ADMINISTRATION
 encapsulation dot1Q 30
 ip address 192.168.30.1 255.255.255.0
 ip nat inside
 ip helper-address 192.168.50.10
 no shutdown
!
interface GigabitEthernet0/1.40
 description VLAN40-GUEST
 encapsulation dot1Q 40
 ip address 192.168.40.1 255.255.255.0
 ip nat inside
 ip helper-address 192.168.50.10
 no shutdown
!
! ── VLAN 50 — Management IT (CORRECTION : NAT inside ajouté)
interface GigabitEthernet0/1.50
 description VLAN50-MANAGEMENT
 encapsulation dot1Q 50
 ip address 192.168.50.1 255.255.255.0
 ip nat inside
 ! NAT inside activé pour permettre à DC-IRIS-01 (192.168.50.10)
 ! d'accéder à Internet (Windows Update, activation, téléchargement rôles)
 no shutdown
!
interface GigabitEthernet0/1.99
 description VLAN99-PRE_AUTH-QUARANTAINE
 encapsulation dot1Q 99
 ip address 192.168.99.1 255.255.255.0
 ip nat inside
 ip helper-address 192.168.50.10
 ! Accès Internet limité (portail captif possible)
 ip access-group PRE_AUTH_FILTER in
 no shutdown
!
! ── NAT Overload (PAT) ────────────────────────────────────
!
! CORRECTION APPLIQUÉE : permit 192.168.50.0 0.0.0.255 ajouté
! pour que le DC-IRIS-01 puisse joindre Internet.
!
ip access-list standard NAT_LIST
 permit 192.168.10.0 0.0.0.255    ! VLAN 10 Étudiants
 permit 192.168.20.0 0.0.0.255    ! VLAN 20 Profs
 permit 192.168.30.0 0.0.0.255    ! VLAN 30 Administration
 permit 192.168.40.0 0.0.0.255    ! VLAN 40 Guest
 permit 192.168.50.0 0.0.0.255    ! VLAN 50 Management IT — CORRECTION RP01
 permit 192.168.99.0 0.0.0.255    ! VLAN 99 PRE_AUTH (accès limité)
 deny any
!
ip nat inside source list NAT_LIST interface GigabitEthernet0/0 overload
!
! ── ACL — Filtrage WAN entrant ────────────────────────────
ip access-list extended BLOCK-WAN-IN
 permit udp any any eq 500       ! IKE/IPSec
 permit esp any any               ! ESP
 permit icmp any 192.168.50.0 0.0.0.255 echo-reply
 deny   tcp any any eq 23 log    ! Telnet
 deny   tcp any any eq 3389 log  ! RDP direct depuis WAN
 permit ip any any
!
! ── ACL — Filtrage PRE_AUTH (VLAN 99) ────────────────────
! Autorise uniquement DNS + DHCP + RADIUS + accès Web limité
! Interdit accès aux VLANs internes (isolation quarantaine)
!
ip access-list extended PRE_AUTH_FILTER
 permit udp any any eq 67          ! DHCP request
 permit udp any any eq 68          ! DHCP reply
 permit udp any host 192.168.50.10 eq 1812  ! RADIUS Auth
 permit udp any host 192.168.50.10 eq 1813  ! RADIUS Accounting
 permit tcp any host 192.168.50.10 eq 53    ! DNS TCP
 permit udp any host 192.168.50.10 eq 53    ! DNS UDP
 permit tcp any any eq 80          ! HTTP (portail captif possible)
 permit tcp any any eq 443         ! HTTPS
 ! Interdire accès aux VLANs internes depuis PRE_AUTH
 deny ip 192.168.99.0 0.0.0.255 192.168.10.0 0.0.0.255 log
 deny ip 192.168.99.0 0.0.0.255 192.168.20.0 0.0.0.255 log
 deny ip 192.168.99.0 0.0.0.255 192.168.30.0 0.0.0.255 log
 deny ip 192.168.99.0 0.0.0.255 192.168.40.0 0.0.0.255 log
 deny ip 192.168.99.0 0.0.0.255 192.168.50.0 0.0.0.255 log
 permit ip any any
!
! ── Routes statiques ──────────────────────────────────────
ip route 0.0.0.0 0.0.0.0 dhcp
!
! ── NTP / Syslog ──────────────────────────────────────────
ntp server 0.fr.pool.ntp.org
logging host 192.168.50.10
logging trap informational
!
! ── SSH + Accès depuis VLAN 50 uniquement ─────────────────
ip domain-name mediaschool.local
crypto key generate rsa modulus 2048
ip ssh version 2
ip ssh time-out 60
ip ssh authentication-retries 3
!
line vty 0 4
 login local
 transport input ssh
 access-class 51 in
!
ip access-list standard 51
 permit 192.168.50.0 0.0.0.255
 deny any log
!
line console 0
 login local
 logging synchronous
!
! ── Désactiver services inutiles ──────────────────────────
no ip http server
no ip http secure-server
no cdp run
!
end
